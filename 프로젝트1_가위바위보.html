model=keras.models.Sequential()
model.add(keras.layers.Conv2D(16, (3,3), activation='relu', input_shape=(28,28,1))) 
#16개의 이미지 특징 # 활성함수(activation)은 relu = rectified linear unit 선형 유닛 
model.add(keras.layers.MaxPool2D(2,2))
#MaxPooling : 해당 영역에서 최댓값을 찾는 방법 
model.add(keras.layers.Conv2D(32, (3,3), activation='relu')) #32개의 이미지 특징 
model.add(keras.layers.MaxPooling2D((2,2)))
model.add(keras.layers.Flatten()) #flatten() : 다차원을 1차원으로 품 
model.add(keras.layers.Dense(32, activation='relu')) #dense 레이어가 클수록 복잡한 분류기를 만들 수 있음
model.add(keras.layers.Dense(10, activation='softmax'))
print('Model에 추가된 Layer 개수: ', len(model.layers))
model.summary()

<본격 학습시키기>

1) 학습 방식에 대한 환경설정 -> compile

  • model함수 : 10개의 숫자 중 어느 것일지에 대한 확률값을 출력하는 함수
  • optimizer (정규화)
  • loss function(손실함수) : 모델 최적화에 사용되는 목적 함수로, categorical_crossentropy or mse와 같은 기존 손실 함수의 문자열을 사용할 수 있음
  • metric(기준) 리스트 : 분류를 하고자 할 때 metrics=['accuracy']로 설정하면 됨
  • fit : 모델을 학습시키기 위해 사용되는 함수
  • fit함수 형식(data, labels, epochs, batch_size,..)

    imgs=np.zeros(number_of_data*img_size*img_size*color,dtype=np.int32).reshape(number_of_data,img_size,img_size,color)
    labels=np.zeros(number_of_data,dtype=np.int32)

        labels[idx]=0 # 가위 : 0

    for file in glob.iglob('C:/Users/User/Desktop/rock_scissor_paper/rock/*.jpg'):

        imgs[idx,:,:,:]=img # 데이터 영역에 이미지 행렬을 복사
        labels[idx]=1 # 바위 : 1
        idx=idx+1
        
      
    for file in glob.iglob('C:/Users/User/Desktop/rock_scissor_paper/paper*.jpg'):
        img = np.array(Image.open(file),dtype=np.int32)
        imgs[idx,:,:,:]=img # 데이터 영역에 이미지 행렬을 복사
        labels[idx]
        
    print("학습데이터(x_train)의 이미지 개수는",idx,"입니다.")
    return imgs, labels
